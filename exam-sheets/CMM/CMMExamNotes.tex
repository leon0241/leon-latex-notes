		\begin{thm}[Stokes' Theorem]{thm:stokes}{ }
			Let $\Sigma$ be a smooth oriented surface in $\mathbb{R}^3$ with boundary $\partial \Sigma \equiv \Gamma$. If a vec field $\bm{F}(x, y, z)$ is defined and has cont. first order partial \\
			derivs in a region containing $\Sigma$, then $\iint_\Sigma (\nabla \times \bm{F}) \cdot \text{ d}\Sigma = \oint_{\partial \Sigma} \bm{F} \cdot \text{ d}\Gamma$, or:
			$\iint_\Sigma\left(\left(\frac{\partial F_z}{\partial y} \text{ dy} \text{ dz}\right) + \left(\frac{\partial F_x}{\partial z} - \frac{\partial F_z}{\partial x}\right) \text{ dz} \text{ dx} + \left(\frac{\partial  F_y}{\partial  x} - \frac{\partial  F_x}{\partial y}\right) \text{ dx} \text{ dy} \right) = \oint_{\partial \Sigma} \left(F_x \text{ dx} + F_y \text{ dy} + F_z \text{ dz}\right)$
			
		\end{thm}

		\begin{dfn}[The Newtonian Universe]{dfn:newtonuni}{ }
			\begin{itemize}
				\item An \textbf{event} is some $(t, \bm{a})\in \mathbb{R} \times \mathbb{R}^3$ labelling a fixed $t$ in time and $\bm{a}$ in space.
				\item Two events $(t_0, \bm{a}), (t_1, \bm{b})$ are \textbf{simultaneous} iff $t_0 = t_1$. The distance between simultaneous events is the norm $\left|\bm{a} - \bm{b}\right|$
				\item A \textbf{particle} is an object whose position in the universe is given at any time by one event
				\item A \textbf{worldline} is a particle's trajectory through the universe. It is a graph $\bm{x} : \mathbb{R} \to \mathbb{R}^3, t \mapsto\bm{x}(t)$ of form $(t, \bm{x}(t))\in \mathbb{R} \times \mathbb{R}^3$.
				\item The \textbf{configuration space} of a \textbf{many-body system of $n$ particles} is the location of each particle in space, so is $\mathbb{R}^3 \times \mathbb{R}^3 \times \dots \times \mathbb{R}^3 = \mathbb{R}^{3n} = \mathbb{R}^N$.
				\item For the worldline of the $a^{th}$ particle denoted $\bm{x}_a(t): \mathbb{R} \to \mathbb{R}^3$, the collection of the worldlines defines a curve $\bm{x}(t) = (\bm{x}_1(t), \bm{x}_2(t), ..., \bm{x}_n(t)): \mathbb{R} \to \mathbb{R}^N$ in the configuration space.
			\end{itemize}
			
		\end{dfn}
		
		
		\begin{dfn}[Newton's Equation]{dfn:newteqn}{ }
			The PoD implies the equation $\bm{\ddot{x}}(t) = \bm{\Phi}(\bm{x}, \bm{\dot{x}}, t)$, $\bm{\Phi}: \mathbb{R}^N \times \mathbb{R}^N \times \mathbb{R} \to \mathbb{R}^N$. For $\bm{v} : \mathbb{R} \to \mathbb{R}^N$, the system $\binom{\bm{\dot{x}}(t)}{\bm{v}(t)} = \binom{\bm{v(t)}}{\bm{\Phi}(\bm{x}, \bm{\dot{x}}, t)}$ has the same soln space.
			
		\end{dfn}
		
		
		\begin{mthm}[Newton's Laws]{thm:newtslaws}{ }
			\begin{enumerate}[label=(\arabic*)]
				\item A system of particles is \textbf{freely-moving} if $\bm{\Phi}(\bm{x}, \bm{\dot{x}}, t) = 0$. Such a system with initial state $\bm{x}(0) = \bm{x}_0, \bm{\dot{x}}(0) = \bm{v}_0$ is solved by $\bm{x}(t) = \bm{x}_0 + t\bm{v}_0$.
				\item A force is a map $\bm{F} : \mathbb{R}^3 \to \mathbb{R}^3$ affecting a particle's velocity. For a particle of inertial mass $m$, $\bm{\Phi}$ is a force field is $\bm{F}(\bm{x}) = m \bm{\ddot{x}}$. A point $\bm{x}\in \mathbb{R}^3$ with $\bm{F}(\bm{x}) = 0$ is an \textbf{equilibrium point} of $\bm{F}$.
			\end{enumerate}
			
		\end{mthm}
		
		
		\begin{dfn}[Affine Transformation]{dfn:affntrans}{ }
			\textbf{Affine} transformations preserve collinearity and ratios of distance
			
		\end{dfn}
		
		
		\begin{dfn}[Galilean Transformations]{dfn:galiltransforms}{ }
			A \textbf{Galilean transformation} is an affine transformation which preserves time intervals and the distance between simultaneous events
			\begin{itemize}
				\item A Galilean transformation is given by a linear transformation of the form $\begin{psmallmatrix*}
					1&\bm{0}&s\\
					\bm{v}&R&\bm{s}\\
					0&\bm{0}&1
				\end{psmallmatrix*}$ where $\bm{s, v}\in \mathbb{R}^3, s\in \mathbb{R},$ and $R: \mathbb{R}^3 \to \mathbb{R}^3$ is an orthogonal transformation.
			\end{itemize}
			
			\includegraphics[scale=0.28]{galiltrans}
			
		\end{dfn}
		
		
		\begin{thm}[Galilean Relativity]{thm:galirel}{ }
			The laws of motion (Newton's Eq.) are invariant under Galilean transformations - translation, fixed $\bm{\dot{x}}$ motion, and rotation of spatial axes.
			
		\end{thm}
