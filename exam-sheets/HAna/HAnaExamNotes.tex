		\begin{dfn}[Bounds and Boundedness]{dfn:bounds}{ }
			Let $A \subseteq R$.
			\begin{itemize}[label=*]
				\item $x \in \mathbb{R}$ is an \textbf{upper bound for $A$} if $x \ge a$ for all $a \in A$
				\item $s \in \mathbb{R}$ is a \textbf{least upper bound} or \textbf{supremum} for $A$ if $s$ is an upper bound for $A$, and $x \ge s$ whenever $x$ is an upper bound for $A$.
				\item $A$ is \textbf{bounded} if it is bounded above and below
			\end{itemize}
			
		\end{dfn}
		
		\begin{axi}[The Completeness Axiom]{axi:completeness}{ }
			Every bounded above non-empty $A \subseteq \mathbb{R}$ has a least upper bound in $\mathbb{R}$
			
		\end{axi}
		
		
		\begin{dfn}[Nested Sets]{dfn:nestsets}{ }
			A sequence $(I_n)_{n \in \mathbb{N}}$ of sets is \textbf{nested} if $I_n \supseteq I_{n+1}$ for all $n \in \mathbb{N}$
			
		\end{dfn}
		
		
		\begin{thm}[The Nested Interval Theorem]{thm:nestints}{ }
			If $(I_n)_{n \in \mathbb{N}}$ is a nested sequence of nonempty closed bounded intervals, then the set $E := \bigcap_{n \in \mathbb{N}}I_n = \left\{x \in \mathbb{R}: x \in I_n \forall n \in \mathbb{N}\right\}$ is nonempty. \\
			If $\lambda(I_n) \to 0$ as $n \to \infty$ then $E$ contains exactly one element.
			
		\end{thm}
		
		
		\begin{dfn}[Sequence Convergence]{dfn:seqcnvgnce}{ }
			A real sequence $(a_n)_{n \in \mathbb{N}}$ \textbf{converges} to $a \in \mathbb{R}$ iff for every $\epsilon > 0$ there exists $N \in \mathbb{N}$ such that $\left|a_n - a\right| <\epsilon$ for all $n \ge N$. A sequence which does not converge to some real number \textbf{diverges}.
			\begin{itemize}
				\item We say $a_n \to \infty$ as $n \to \infty$ if $\forall R \in \mathbb{R}$, $\exists N \in \mathbb{N}$ s.t. $a_n > R$ for all $n \ge \mathbb{N}$
				\item We say $a_n \to -\infty$ as $n \to \infty$ if $\forall R \in \mathbb{R}$, $\exists N \in \mathbb{N}$ s.t. $a_n < R$ for all $n \ge \mathbb{N}$
			\end{itemize}
			
		\end{dfn}
		
		
		\begin{mdfn}[Cauchy Sequences]{mdfn:cauchyseq}{ }
			A real sequence $(a_n)_{n \in \mathbb{N}}$ is \textbf{Cauchy} if for every $\epsilon > 0$ there exists $N \in \mathbb{N}$ s.t. $\left|a_n - a_m\right| <\epsilon$ for all $n, m \ge N$.\\
			
		\end{mdfn}
		
		
		\begin{mthm}[The Cauchy Criterion]{thm:}{ }
			A real sequence is convergent if and only if it is Cauchy.
			\vspace{1pt}
			\hrule	
			\vspace{-6pt}
			\begin{proof}
				1: Let $(a_n)_{n\in \mathbb{N}}$ be convergent. Then $\exists a\in\mathbb{R}$ s.t. $\lim_{n\to \infty}a_n = a$, and for all $ \epsilon > 0$, $\exists N\in \mathbb{N}$ s.t. $\left|a_n -a\right| <\epsilon/2$ when $n \ge N$. Then by the TI, $\left|a_n - a_m\right| = \left|(a_n-a) + (a-a_m)\right| \le \left|a_n - a\right| + \left|a-a_m\right| = \left|a_n - a\right| + \left|a_m -a\right| <\epsilon/2 +\epsilon/2 =\epsilon$ for $m, n \ge N$, thus the sequence is Cauchy.\\
				2: Let $(a_n)_{n\in \mathbb{N}}$ be Cauchy. First show it is bdd. Let $\epsilon = 1$. Then $\exists N\in \mathbb{N}$ s.t. $\left|a_n -a_m\right| < 1$ for $m, n \ge N$. In particular $\left|a_m\right| < 1+\left|a_N\right|$ for all $m \ge N$, so $\left|a_m\right| \le \max\{\left|a_1\right|, ..., \left|a_{N-1},\right|, \left|a_N\right| + 1\}$ for $m\in \mathbb{N}$, so $(a_n)_{n\in \mathbb{N}}$ is bdd. Secondly, by the B-WT, $\exists (a_{n_k})_{k\in \mathbb{N}}$ s.t. $a_{n_k} \to a\in \mathbb{R}$ as $k \to \infty$. Finally, show $a_n \to a$: Let $\epsilon > 0$. Since $(a_n)_{n\in \mathbb{N}}$ is Cauchy, $\exists N, K\in \mathbb{N}$ s.t. $\left|a_n - a_m\right| <\epsilon/2$ for $m, n \ge N$ and $\left|a_{n_k} - a \right| <\epsilon/2$ for $k \ge K$. Fix $k\in \mathbb{N}$ s.t. $k\ge K$ and $n_k \ge N$. Then $\left|a_n - a\right| \le \left|a_n - a_{n_k}\right| + \left|a_{n_k} - a\right| <\epsilon/2 +\epsilon/2 =\epsilon$, $\forall n \ge N$. Thus $a_n \to a$ as $n \to \infty$ as required.
			\end{proof}
			
		\end{mthm}
		
		
		\begin{dfn}[Subsequences]{dfn:subseqs}{ }
			Let $(a_n)_{n \in \mathbb{N}}$ be a real sequence. A \textbf{subsequence} of $(a_n)_{n \in \mathbb{N}}$ is a real sequence of the form $(a_{n_k})_{k \in \mathbb{N}}$ where for each $k \in \mathbb{N}$ there exists $n_k \in \mathbb{N}$ such that $n_1 < n_2< ... < n_k < n_{k+1} < ...$.
			
		\end{dfn}
		
		
		\begin{thm}[Bolzano-Weierstrass Theorem]{thm:bolzweier}{ }
			Every bounded sequence of real numbers has a convergent subsequence.
			\vspace{1pt}
			\hrule	
			\vspace{-6pt}
			\begin{proof}
				Let $(a_n)_{n\in \mathbb{N}}$ be a bdd seq., with upper/lower bound $a, b\in \mathbb{R}$. Let $I_1 = [a, b]$ and divide $I_1$ into $I_1' = [a, (a+b)/2], I_1'' = [(a+b)/2, b]$. Since $I_1 = I_1' \cup I_1''$, at least one of $I_1', I_1''$ contains infinitely many $a_n$. Choose one with this property and denote it $I_2$. Let $n_2\in \mathbb{N}$ be s.t. $n_2 > n_1 = 1$ and $a_{n_2}\in I_2$. Repeat inductively.\\
				By the above process we recursively define a nested sequence of intervals $(I_k)_{k\in \mathbb{N}}$ and subsequence $(a_{n_k})_{k\in \mathbb{N}}$ s.t. each $I_k$ contains infinitely many terms of $(a_n)_{n\in \mathbb{N}}$, $a_{n_k}\in I_k$, and $\lambda(I_k) = 2^{-(k-1)}(b-a)$.\\
				Suppose in this way we construct $a_{n_1}, ..., a_{n_k}, k\in \mathbb{N}$, and intervals $I_1, ..., I_k$. Divide $I_k$ into $I_k', I_k''$ as above, again at least one contains infinitely many $a_n$. Choose one with this property, denote it $I_{k +1}$. Let $n_{k + 1}\in \mathbb{N}$ be s.t. $n_{k +1} > n_k$, $a_{N_{k+1}}\in I_{k+1}$.\\
				Now observe $(I_k)_{k\in \mathbb{N}}$ is a nested seq. of clounded intervals, so by the NIT there is $x\in \mathbb{R}$ s.t. $x$ lies in all $I_k$. Since $a_{n_k}\in I_k$, $\left|x - a_{n_k}\right| \le \lambda(I_k) = \frac{b-a}{2^{k-1}}$ for all $k\in \mathbb{N}$. Then by the Squeeze Theorem, $a_{n_k} \to x$ as $k \to \infty$.
			\end{proof}
			
		\end{thm}
		
		
		
		{\tiny \textbf{\underline{Series}}}
		
		
		
		\begin{mdfn}[Infinite Series]{dfn:infsers}{ }
			Let $(a_k)_{k \in N}$ be a real sequence. We form the \textbf{sequence of partial sums $(s_n)_{n \in \mathbb{N}}$} by $s_n := \sum_{k=1}^{n} a_k$ for all $n \in\mathbb{N}$.
			\begin{itemize}
				\item $\sum_{k=1}^{\infty}a_k$ \textbf{converges to $a$} if $\lim_{n\to \infty} s_n$ exists and $= a$, else \textbf{diverges}.
			\end{itemize} 
			
		\end{mdfn}
		
		
		\begin{dfn}[Absolute Convergence]{dfn:absconv}{ }
			Let $(a_n)_{n \in \mathbb{N}}$ be a real sequence. Then $\sum_{n=1}^{\infty} a_n$:
			\begin{enumerate}[label=(\arabic*)]
				\item \textbf{Converges absolutely} if $\sum_{n=1}^{\infty} \left|a_n\right|$ converges
				\item \textbf{Converges conditionally} if it converges but $\sum_{n=1}^{\infty}\left|a_n\right|$ diverges.
			\end{enumerate}
			
		\end{dfn}
		
		
		\begin{thm}[Absolute Convergence vs Convergence]{thm:absconvvsconv}{ }
			Absolute convergence implies convergence. Note the converse does not always hold (consider $\sum_{n=0}^{\infty} (-1)^n/n$.
			\vspace{1pt}
			\hrule	
			\vspace{-6pt}
			\begin{proof}
				Let $(a_k)_{k=1}^\infty$ be a real seq., and $\sum_{n=1}^{\infty}a_n$ be absolutely convergent. We show the series satisfies the Cauchy criterion: Let $\epsilon > 0$. Since $\sum_{n=1}^\infty$ cnvges absolutely, by the Cauchy criterion for series $\exists N\in \mathbb{N}$ s.t. $\sum_{k=m+1}^{n}\left|a_k\right| <\epsilon$ for all $n > m \ge N$. Now let $n > m \ge N$ and apply the TI repeatedly to obtain $\left|\sum_{k=m+1}^n a_k\right| \le \sum_{k=m+1}^n \left|a_k\right| <\epsilon$. This establishes the Cauchy criterion for $\sum_{n=1}^\infty a_n$, so the series is convergent.
			\end{proof}
			
		\end{thm}
		
		
		\begin{mthm}[The Cauchy Criterion For Series]{thm:cauchforseries}{ }
		Let $(a_k)_{k=1}^\infty$ be a real sequence. Then $\sum_{k=1}^{\infty}a_k$ converges iff for all $\epsilon > 0$ there exists $N \in \mathbb{N}$ such that $\left|\sum_{k=m+1}^{n}a_k\right| <\epsilon$ for all $n > m \ge N$.
		\vspace{1pt}
		\hrule	
		\vspace{-6pt}
		\begin{proof}
			Let $(s_n)_{n\in \mathbb{N}}$ be the seq. of partial sums of the series. Observe that $s_n - s_m = \sum_{k=1}^{n}a_k - \sum_{k=1}^{m}a_k = \sum_{k=m+1}^{n}a_k$ for $n > m$. Then the series satisfies the Cauchy criterion iff $(s_n)_{n\in \mathbb{N}}$ is Cauchy. $(s_n)_{n\in \mathbb{N}}$ is Cauchy iff it is convergent, and convergent iff $\sum_{k=1}^{\infty}a_k$ converges.
		\end{proof}
		
		\end{mthm}
		
		
		\begin{thm}[The Riemann Rearrangement Theorem]{thm:riearrange}{ }
			Let $(a_n)_{n \in \mathbb{N}}$ be a real sequence and suppose $\sum_{n=1}^{\infty}a_n$ is conditionally convergent. Then for \textit{any} $s \in \mathbb{R}$, there exists a bijection $\sigma: \mathbb{N} \to \mathbb{N}$ such that the rearrangement $\sum_{n=1}^{\infty}a_{\sigma(n)}$ converges to $s$. 
			\vspace{1pt}
			\hrule
			\vspace{2pt}
			Note the RRT does not apply to absolutely convergent series. Indeed, for $\sum_{n=1}^{\infty}a_n$ absolutely convergent, and any bijection $\sigma: \mathbb{N} \to \mathbb{N}$, $\sum_{n=1}^{\infty}a_{\sigma(n)}$ is convergent and $\sum_{n=1}^{\infty}a_{\sigma(n)} = \sum_{n=1}^{\infty}a_n$
			
		\end{thm}
		
		
		
	{\tiny \textbf{\underline{Subsets Of The Real Line}}}
		
		
		
		\begin{dfn}[Open Sets]{dfn:opensets}{ }
			$U \subseteq \mathbb{R}$ is \textbf{open} if for all $x \in U$ there exists $r > 0$ s.t. $(x-r, x+r) \subseteq U$.
			
			\begin{itemize}
				\item The union of any collection of open sets is open
				\item The intersection of any finite collection of open sets is open
				\item Any nonempty open $U \subseteq \mathbb{R}$ is a finite or countable union of nonempty disjoint open intervals.
				\item If $f: \mathbb{R} \to \mathbb{R}$ is continuous and $\lambda \in \mathbb{R}$, then $\{x \in \mathbb{R} : f(x) > \lambda\}$ is open.
				\item For $U \subseteq\mathbb{R}$, $f: U \to \mathbb{R}$ cont., $\lambda\in \mathbb{R}$, $\left\{x\in U : f(x) > \lambda\right\}$ is open.
			\end{itemize}
			
		\end{dfn}
		
		
		\begin{dfn}[Closed Sets]{dfn:closedsets}{ }
			A set $F \subseteq \mathbb{R}$ is \textbf{closed} if its complement $\mathbb{R} \backslash F$ is open.
			
			\begin{itemize}
				\item The union of any finite collection of closed sets is closed
				\item The intersection of any collection of closed sets is closed
				\item If $f: \mathbb{R} \to \mathbb{R}$ is continuous and $ \lambda \in \mathbb{R}$, then $\{x \in \mathbb{R} : f(x) \le \lambda\}$ is closed.
			\end{itemize}
			
		\end{dfn}
		
		
		\begin{dfn}[Sequential Closure]{dfn:seqclos}{ }
			$F \subseteq \mathbb{R}$ is \textbf{sequentially closed} if for every convergent sequence $(x_n)_{n \in \mathbb{N}}$ s.t. $x_n \in F$ for all $n \in \mathbb{N}$, the limit $x := \lim_{n\to \infty}x_n$ satisfies $x \in F$.
			\begin{itemize}
				\item $F \subseteq \mathbb{R}$ is closed if and only if it is sequentially closed.
			\end{itemize}
			\vspace{1pt}
			\hrule	
			\vspace{-6pt}
			\begin{proof}
				First prove by contrapositive closed sets are sequentially closed: Let $F$ be closed and $(x_n)_{n\in \mathbb{N}}$ convergent s.t. $x:= \lim_{n\to \infty}x_n \notin F$. We show $\exists N\in \mathbb{N}$ s.t. $x_N \notin F$. $F$ is closed, so by definition $U := \mathbb{R} \backslash F$ is open. \\
				Moreover $x\in U$, so $\exists\epsilon > 0 $ s.t. $(x-\epsilon, x+\epsilon \subseteq U)$, and $N\in \mathbb{N}$ s.t. \\
				$\left|x_n - x\right| <\epsilon$ $\forall n\in \mathbb{N}$. In particular $x_N\in (x-\epsilon, x +\epsilon) \subseteq U$, so $x_N \notin F$.\\
				Now prove by contradiction sequentially closed sets are closed: Suppose $F$ is not closed, equivalently $U := \mathbb{R} \backslash F$ is not open. Then $\exists x\in U$ s.t. $\forall r > 0$, $(x-r, x+r) \cap F \ne \emptyset$. For $n\in \mathbb{N}$, let $x_n \in ((x - \frac{1}{n}, x+ \frac{1}{n}) \cap F)$. Then $(x_n)_{n\in \mathbb{N}}$ converges to $x$, since $\left|x - x_n\right| \le \frac{1}{n}$. In particular, since $F$ is sequentially closed, we must have $x\in F$, a contradiction.
			\end{proof}
			
		\end{dfn}
		
		
		\begin{dfn}[The Middle Thirds Cantor Set]{dfn:midthrdscantor}{ }
			The \textbf{Middle Thirds Cantor Set $C$} is a closed set defined as follows: 
			\begin{enumerate}[label=(\arabic*)]
				\item Divide $[0, 1]$ into thirds, remove the open middle third interval $(\frac{1}{3}, \frac{2}{3})$.
				\item Divide the remaining two intervals into thirds and remove the open middle thirds intervals $(\frac{1}{9}, \frac{2}{9})$ and $(\frac{7}{9}, \frac{8}{9})$.
				\item Repeat this process ad infinitum.
			\end{enumerate}
			At the $n^{th}$ stage of the construction we have $2^n$ closed intervals $(I_{n, k})_{k=1}^{2n}$ with $\lambda(I_{n, k}) = 3^{-n}$ for all $1 \le k \le 2^n$. Then $C := \bigcap_{n =1}^{\infty}C_n$ where $C_n :+ \bigcup_{k =1}^{2^n}I_{n, k}$ for $n \in \mathbb{N}$. Each $I_{n, k}$ is closed, so their finite union $C_n$ and the intersection $C$ of these $C_n$ are also closed.
			
		\end{dfn}
		
		
		\begin{xmp}[Showing Not Closed And Not Open]{xmp:notclsdoropn}{ }
			Show $S := \{e^k : k\in \mathbb{Z}\}$ is neither open nor closed.\\
			First show $S$ is not open: Let $k=0$. $e^0 = 1$, so $1\in S$. For $k > 0$, $e^k \ge e^1$, and for $k < 0, e^k \le e^{-1}$. Thus $S \cap (e^{-1}, e) = \{1\}$. Since $\{1\}$ is not open, $S$ cannot be open, as $(e^{-1}, e)$ is open and a finite intersection of open sets is open.\\
			Now show $S$ is not closed: The seq $x_n := e^{-n}$ has $x_n\in S$ for all $n\in \mathbb{N}$ but $\lim_{n\to \infty} x_n = 0 \notin S$. Thus $S$ is not sequentially closed, ie not closed.
			
		\end{xmp}
		
		
		\begin{xmp-s}[A Seq With Bounded Term Differences Is Cauchy]{xmp:bddtermdiffsmeanscauch}{ }
			Let $(a_n)_{n\in \mathbb{N}}$ be s.t. $\left|a_n - a_{n-1}\right| \le 2^{-n}$. Show $(a_n)_{n\in \mathbb{N}}$ is Cauchy.\\
			Consider $\left|a_n - a_m\right|$. Form the telescoping sum then apply the TI: $\left|a_n - a_m\right| \le \left|a_n - a_{n-1}\right| + \left|a_{n-1} - a_{n-2}\right| + ... + \left|a_{m+1} - a_m\right| \le \frac{1}{2^n} + \frac{1}{2^{n-1}} + \dots + \frac{1}{2^{m+1}}$. Thus $\left|a_n - a_m\right| \le \sum_{k=m+1}^{n} \frac{1}{2^k} \le \frac{1}{2^m}$, so given $\epsilon > 0$ choose $N\in \mathbb{N}$ s.t. $\frac{1}{2^N} <\epsilon$ and the seq is Cauchy.
			
		\end{xmp-s}

	{\footnotesize \textbf{\underline{Measure And Integration}}}
	
	
	{\tiny\textbf{\underline{The Lebesgue Measure}}}
		
		
		
		\begin{dfn}[Length]{dfn:length}{ }
			For a nonempty bounded interval $I \subseteq \mathbb{R}$, we define the \textbf{length $\lambda$ of I} as $\lambda(I) = \sup(I) - \inf(I)$. $\lambda(\emptyset) = 0$ and for $I$ unbounded, $\lambda(I) = \infty$.
			
		\end{dfn}
		
		
		\begin{mdfn}[Lebesgue Measurable Sets]{mdfn:lebmesable}{ }
			$E \subseteq \mathbb{R}$ is \textbf{Lebesgue measurable} if for all $\epsilon > 0$, there exists open $U \subseteq \mathbb{R}$ s.t. $E \subseteq U$ and $m(U \backslash E) <\epsilon$. The collection of all L-mesable subsets of $\mathbb{R}$ is $\mathcal{M} \subseteq \mathcal{P}(\mathbb{R})$.\\
			We only handle L-mesable sets in this course thankfully!
			
		\end{mdfn}
		
		\begin{mdfn}[The Lebesgue Measure]{dfn:lebmes}{ }
			The \textbf{Lebesgue measure} is the function $m : \mathcal{M} \to [0, \infty]$. For $E\in \mathcal{M}$, $m(E) := \inf \left\{\sum_{j=1}^{\infty}\lambda(I_j) : E \subseteq\bigcup_{j=1}^\infty I_j, I_j \subseteq \mathbb{R} \text{ an open interval}\right\}$. Here the inf is over all countable coverings $(I_j)_{j\in \mathbb{N}}$ of $E$ by open intervals.
			\begin{itemize}
				\item[*] $m(\emptyset) = 0$
				\item[*] If $E \subseteq F \subseteq \mathbb{R}$, $m(E) \le m(F)$
				\item[*] If $(E_j)_{j\in \mathbb{N}}$ is any seq of sets in $\mathcal{M}$, $m\left(\bigcup_{j=1}^{\infty} E_j\right) \le \sum_{j=1}^{\infty}m(E_j)$.
				\item[*] If the $E_j$ above are all disjoint then $m\left(\bigcup_{j=1}^{\infty} E_j\right) = \sum_{j=1}^{\infty}m(E_j)$
				\item If $\sum_{j=1}^{\infty}\lambda(I_j)$ diverges, we say it has value $\infty$. If there is no cover of $E$ s.t. $\sum_{j=1}^{\infty}\lambda(I_j) < \infty$, $m(E) = \infty$.
				\item For any interval $J \subseteq\mathbb{R}$, $\lambda(J) = m(J)$.
				\item Given $E \subseteq\mathbb{R}$, $\epsilon > 0$, $\exists$ open $U \subseteq \mathbb{R}$ s.t. $E \subseteq U$ and $m(U) < m(E) +\epsilon$.
			\end{itemize}
			
		\end{mdfn}
		
		
		\begin{dfn}[The Lebesgue Outer Measure]{dfn:leboutmes}{ }
			The \textbf{Lebesgue Outer Measure} is $m^* : \mathcal{P} \to [0, \infty]$. We base our definition of $m$ on this, $m^*(E) = m(E)$ for $E\in \mathcal{M}$. For $E \notin \mathcal{M}$ it is only countably subadditive (ie measure of union $\le$ sum of measures).
			
		\end{dfn}
		
		
		\begin{dfn}[Null Sets]{dfn:nullsets}{ }
			A set $E \subseteq \mathbb{R}$ is \textbf{null} or \textbf{has measure zero} if $m(E) = 0$.\\
			Every countable subset of $R$ has measure zero.
			\vspace{1pt}
			\hrule	
			\vspace{-6pt}
			\begin{proof}
				First prove any singleton set $\{x\}$ for $x\in \mathbb{R}$ has measure zero. Given $x\in \mathbb{R}$, let $\epsilon > 0$ and consider the cover $I  := \left(x-\epsilon/2, x +\epsilon/2\right)$. Then $m(\{x\}) \le \lambda(I) =\epsilon$. Since $\epsilon > 0$ is arbitrary we conclude $m(\{x\}) = 0$. \\
				Any countable $E$ is a countable union of singletons, $E = \bigcup_{j=1}^\infty \{x_j\}$. Apply the countable subadditivity property to conclude $m(E) = 0$.
			\end{proof}
			
		\end{dfn}
		
		
		\begin{dfn}[Distance Between Sets]{dfn:dist}{ }
			For $E, F \subseteq \mathbb{R}$ $\ne \emptyset$ and disjoint; dist$(E, F) := \inf\left\{\left|x-y\right|: x\in E, y\in F\right\}$.
			\begin{itemize}
				\item If $E, F \subseteq \mathbb{R}$ $\ne \emptyset$, disjoint, dist$(E, F) > 0$; $m*(E \cup F) = m(E) + m(F)$
				\item If $F, K \subseteq \mathbb{R}$ nonempty, closed, $F \cap K = \emptyset$, $K$ bounded; then dist$(F, K) > 0$.
				\item If $(K_j)_{j\in \mathbb{N}}$ a seq of disjoint closed $K_i$; $m\left(\bigcup_{j=1}^\infty K_j\right) = \sum_{j=1}^{\infty}m(K_j)$
			\end{itemize}
			
		\end{dfn}
		
		
		\begin{thm}[Properties Of L-Measurable Sets]{thm:lmesprops}{ }
			\begin{itemize}
				\item Any open set is L-measurable
				\item Let $F \subseteq \mathbb{R}$ and $(E_j)_{j\in \mathbb{N}}$ be L-mesable. Then:
					\begin{enumerate}[label=(\arabic*)]
						\item The union $E := \bigcup_{j=1}^\infty E_j$ is L-mesable
						\item The complement $\mathbb{R} \backslash F$ is L-mesable
						\item The intersection $\bigcap_{j=1}^\infty E_j$ is L-mesable
					\end{enumerate}
				\item Let $E, F \subseteq \mathbb{R}$ be L-mesable. Then $F \backslash E$ is L-mesable
				\item For $E \subseteq \mathbb{R}$, $E$ is L-mesable iff $\forall \epsilon > 0$, $\exists$ closed $K \subseteq E$ s.t. $m(E \backslash K) <\epsilon$.
			\end{itemize}
			
		\end{thm}
		
		
		\begin{thm}[Monotone Convergence For Measure]{thm:montonconvmes}{ }
			Let $(E_j)_{j\in \mathbb{N}}$ be a seq. of L-mesable sets s.t. $E_j \subseteq E_{j+1}$ $\forall j\in \mathbb{N}$. \\
			Then $m\left(\bigcup_{j=1}^\infty E_j\right) = \lim_{j\to \infty}m(E_j)$
			
		\end{thm}
		
		
		
	{\tiny\textbf{\underline{The Lebesgue Integral}}}
		
		
		
		\begin{dfn}[Characteristic Functions Of Sets]{dfn:setcharfunc}{ }
			For $E \subseteq \mathbb{R}$, the \textbf{characteristic function of $E$} is \\
			$\chi_E: \mathbb{R}\to \mathbb{R}$, $\chi_E(x) := 1$ if $x\in E$; $:= 0$ if $x \notin E$.
			
		\end{dfn}
		
		
		\begin{dfn}[Simple Functions]{dfn:simpfuncs}{ }
			$s: \mathbb{R} \to \mathbb{R}$ is \textbf{simple} if it is a linear combination \\
			of characteristic functions of L-mesable sets. Ie, $\exists c_1, ..., c_k\in \mathbb{R}$ and L-mesable $E_1, ..., E_k \subseteq \mathbb{R}$ s.t. $s(x) = \sum_{j=1}^{k}c_j \chi_{E_j}(x)$ for all $x\in \mathbb{R}$.
			
		\end{dfn}
		
		
		\begin{dfn}[L-Integral Of A Non-Negative Simple Function]{dfn:lintnonnegsimpfunc}{ }
			Let $s: \mathbb{R} \to [0, \infty)$ be simple, $a_j\in [0, \infty), E_j\in \mathcal{M}$, $s = \sum_{j=1}^{k}a_j \chi_{E_j}$. The \textbf{Lebesgue integral of $s$} is $\int_\mathbb{R} s(x) \text{ dx} := \sum_{j=1}^{k}a_j m(E_j)\in [0, \infty)$.
			\begin{itemize}
				\item $\int_\mathbb{R} \chi_E(x) \text{ dx} := m(E)$
				\vspace{1pt}
				\item For $E_j$ with infinite measure, the integral takes value $\infty$. If $a_j = 0$ and $\lambda(E_j) = \infty$, use the convention $0 \cdot \infty = 0$.
			\end{itemize}
			
		\end{dfn}
		
		
		\begin{mdfn}[Lebesgue Measurable Functions]{dfn:lmesablefuncs}{ }
			Let $E \subseteq \mathbb{R}$ be L-mesable. $f: \mathbb{E} \to \mathbb{R}$ is \textbf{Lebesgue measurable} if for all $\lambda\in \mathbb{E}$, the strict superlevel set $\{x\in \mathbb{E} : f(x) > \lambda\}$ is L-mesable
			\begin{itemize}
%				\item Any simple function is L-mesable
				\item Any continuous function $f: \mathbb{R} \to \mathbb{R}$ is L-mesable
				\item If $E \subseteq \mathbb{R}$, $f, g : \mathbb{R} \to \mathbb{R}$ are all L-mesable; then $\max\{f, g\}$, $\min\{f, g\}$, $f\cdot\chi_E$ are all L-mesable
				\item If $f, g : \mathbb{R} \to \mathbb{R}$ L-mesable, $\alpha, \beta\in \mathbb{R}$; then $\alpha f + \beta g$, $f \cdot g$ are L-mesable.
			\end{itemize}
			
		\end{mdfn}
		
		
		\begin{thm}[Approxing Non-Negative L-Measurable Functions]{thm:approxnonneglmesfuncs}{ }
			Let $f: \mathbb{R} \to [0, \infty)$ be a non-negative L-mesable func. Then there exists a sequence $s_n : \mathbb{R} \to [0, \infty)$ of non-negative simple functions s.t.:
			\begin{enumerate}[label=(\arabic*)]
				\item $(s_n)_{n\in \mathbb{N}}$ is ptwise non-decreasing - $0 \le s_n \le s_{n+1}$ for $x\in \mathbb{R}$, $n\in \mathbb{N}$.
				\item $s_n \to f$ pointwise on $\mathbb{R}$ as $n \to \infty$.
			\end{enumerate}
			
		\end{thm}
		
		
		\begin{dfn-s}[Integrating Non-Negative L-Measurable Functions]{dfn:intnonneglmesfuncs}{ }
			Let $f: \mathbb{R} \to [0, \infty)$ be a non-negative L-mesable func. The \textbf{Lebesgue integral of $f$} as $\int_{\mathbb{R}}f(x) \text{ dx} := \sup\left\{\int_{\mathbb{R}}s(x) \text{ dx} : s \text{ simple }, 0 \le s \le f\right\}$ where the supremum is taken over all non-negative simple functions $s: \mathbb{R} \to \mathbb{R}$ with $s(x) \le f(x)$ for all $x\in \mathbb{R}$.
			\begin{itemize}
				\item If $f: \mathbb{R} \to [0, \infty)$, $E \subseteq \mathbb{R}$ both L-mesable, $\int_E f(x) \text{ dx} := \int_{\mathbb{R}} f(x) \chi_E(x) \text{ dx}$
				\item For L-mesable $f$ defined only on L-mesable $E \subseteq \mathbb{R}$, define $\int_E f(x) \text{ dx} = \int_{\mathbb{R}}\tilde{f(x)} \text{ dx}$, with $\tilde{f}: \mathbb{R}\to [0,\infty)$, $\tilde{f}(x) := f(x)$ if $x\in E$, $0$ if $x\in \mathbb{R} \backslash E$.
				\item Let $f, g : \mathbb{R} \to [0, \infty)$ be L-mesable. Then:
					\begin{enumerate}[label=(\arabic*)]
						\item If $0 \le f(x) \le g(x)$ for all $x\in \mathbb{R}$, $0 \le \int_{\mathbb{R}}f(x) \text{ dx} \le \int_{\mathbb{R}}g(x) \text{ dx}$
						\item For all $\lambda > 0$, $m\left(\{x\in \mathbb{R} : f(x) > \lambda\}\right) \le \frac{1}{\lambda} \int_{\mathbb{R}}f(x) \text{ dx}$ (Markov's Ineq.)
					\end{enumerate} 
				\item This L-integral respects linearity
				\item If $f: \mathbb{R} \to [0, \infty)$ is L-mesable and $E, F \subseteq \mathbb{R}$ are disjoint L-mesable sets, $\int_{E \cup F} f(x) \text{ dx} = \int_E f(x) \text{ dx} + \int_F f(x) \text{ dx}$
				\item Let $h_k : \mathbb{R} \to [0, \infty)$ be a seq of non-neg L-mesable funcs on $\mathbb{R}$ s.t. $\sum_{k=1}^{\infty}h_k$ cnvges ptwise. Then $\sum_{k=1}^{\infty} \int_{\mathbb{R}}h_k(x) \text{ dx} = \int_{\mathbb{R}}\sum_{k=1}^{\infty}h_k(x) \text{ dx}$.
			\end{itemize}
			
		\end{dfn-s}
		
		
		\begin{thm}[Pointwise Limits And L-Integration]{thm:ptwiselimslebints}{ }
			The pointwise limit of L-integrals is not necessarily the L-integral of the pointwise limit. Ie, $\exists$ L-mesable $f_n, f: \mathbb{R} \to [0, \infty)$ s.t.:
			\begin{itemize}
				\item $f_n$ is L-integrable for all $n\in \mathbb{N}$ and $f$ is L-integrable
				\item $f_n \to f$ pointwise on $(0, 1]$
				\item $\int_{0}^{1}f(x) \text{ dx} \ne \lim_{n\to \infty} \int_{0}^{1}f_n(x) \text{ dx}$, ie $\lim_{n\to \infty}\int_{0}^{1}f_n(x) \text{ dx} \ne \int_{0}^{1}\left(\lim_{n\to \infty}f_n(x) \right) \text{ dx}$
				\item For counterexample: $f, f_n: \mathbb{R}\to [0, \infty)$, $f(x) := \mathbb{R} \to [0, \infty)$, $f_n(x) := n$ if $0 < x \le \frac{1}{n}$ and $0$ otherwise.
			\end{itemize}
			
		\end{thm}
		
		
		\begin{mthm}[The MCT For Lebesgue Integrals]{thm:mctlebints}{ }
			Let $f_k : \mathbb{R} \to [0, \infty)$ be a ptwise non-decreasing seq. of non-negative L-mesable funcs converging ptwise to $f:=\lim_{k\to \infty} f_k : \mathbb{R} \to [0, \infty)$. Then $f$ is L-mesable, and $\lim_{k\to \infty}\int_{\mathbb{R}}f_k(x) \text{ dx} = \int_{\mathbb{R}}\lim_{k\to \infty}f_k(x) \text{ dx} = \int_{\mathbb{R}}f(x) \text{ dx}$. Note both sides may equal $\infty$.
			
			\begin{itemize}
				\item For $f, g: \mathbb{R} \to [0, \infty)$ L-mesable and $\alpha, \beta \ge 0$, the integral respects linearity
				\item If $f:\mathbb{R} \to [0, \infty)$ L-mesable and $E, F \subseteq \mathbb{R}$ disjoint, $\int_{E \cup F} f(x) \text{ dx} = \int_E f(x) \text{ dx} + \int_F f(x) \text{ dx}$.
				\item For $h_k : \mathbb{R} \to [0, \infty)$ a seq. of L-mesable funcs s.t. $\sum_{k=1}^{\infty} h_k$ converges pointwise, $\sum_{k=1}^{\infty} \int_{\mathbb{R}}h_k(x) \text{ dx} = \int_{\mathbb{R}} \sum_{k=1}^{\infty} h_k(x) \text{ dx}$.
			\end{itemize}
			
		\end{mthm}
		
		
		\begin{lma}[The Fatou Lemma]{lma:fatou}{ }
			If $f_k : \mathbb{R} \to [0, \infty)$ is a seq of non-neg L-mesable funcs, then $\int_{\mathbb{R}}\liminf_{k\to \infty} f_k(x) \text{ dx} \le \liminf_{k\to \infty}\int_{\mathbb{R}}f_k(x) \text{ dx}$.
			
		\end{lma}
		
		
		\begin{lma}[The Reverse Fatou Lemma]{lma:revfatou}{ }
			Let $f_k: \mathbb{R} \to [0, \infty)$ be a seq of non-neg L-mesable funcs and $g: \mathbb{R} \to [0, \infty)$ L-intable s.t. $f_k(x) \le g(x)$ for all $x\in \mathbb{R}$. Then $\int_{\mathbb{R}} \limsup_{k \to \infty} f_k(x) \text{ dx} \ge \limsup_{k \to \infty}\int_{\mathbb{R}}f_k(x) \text{ dx}$.
			
		\end{lma}
		
		
		\begin{dfn-s}[Decomposing L-Measurable Real-Valued Functions]{dfn:decomplmesablefs}{ }
			Let $f : \mathbb{R} \to \mathbb{R}$ be L-mesable. $f^+ (x) := \max\{f(x), 0\}$ and $f^-(x) := -\min\{f(x), 0\}$, $f = f^+ - f^-$, and $\left|f\right| = f^+ + f^-$ are all L-mesable. Moreover, $\int_{\mathbb{R}}\left|f(x)\right| \text{ dx} < \infty \iff \int_{\mathbb{R}}f^+ (x) \text{ dx} < \infty$ and $ \int_{\mathbb{R}}f^- (x) \text{ dx} < \infty$.
			
		\end{dfn-s}
		
		
		\begin{mdfn-s}[Lebesgue Integration Of Real-Valued Functions]{mdfn:lebintsrfuncs}{ }
			An L-mesable $f: \mathbb{R} \to \mathbb{R}$ is \textbf{Lebesgue integrable} if $\int_{\mathbb{R}}\left|f(x)\right| \text{ dx} < \infty$.\\
			The collection of all L-integrable functions is $\mathcal{L}^1(\mathbb{R})$.\\
			For L-integrable $f: \mathbb{R} \to \mathbb{R}$, $\int_{\mathbb{R}}f(x) \text{ dx} := \int_{\mathbb{R}}f^+ (x) \text{ dx} - \int_{\mathbb{R}}f^- (x) \text{ dx}$.
			\begin{itemize}
				\item For $f\in \mathcal{L}^1(\mathbb{R})$, $\left|\int_{\mathbb{R}}f(x) \text{ dx}\right| \le \int_{\mathbb{R}}\left|f(x) \right|dx$
				\item This L-integral respects linearity for $\alpha, \beta\in \mathbb{R}$.
				\item For L-mesable $E \subseteq \mathbb{R}$, $\mathcal{L}^1(E) = \left\{f: E \to \mathbb{R} : \text{$f$ is L-mesable, }\tilde{f}\in \mathcal{L}^1(\mathbb{R})\right\}$.
			\end{itemize}
			
		\end{mdfn-s}
		
		
		\begin{mthm}[The Dominated Convergence Theorem]{mthm:domconv}{ }
			Let $f_k : \mathbb{R} \to \mathbb{R}$ be a seq. of L-integrable funcs converging ptwise to $f: \mathbb{R} \to \mathbb{R}$, and $g\in \mathcal{L}^1(\mathbb{R})$ be s.t. $\left|f_k(x)\right| \le g(x)$ for all $x\in \mathbb{R}, k\in \mathbb{N}$. Then $f\in \mathcal{L}^1(\mathbb{R})$ and $\lim_{k\to \infty}\int_{\mathbb{R}}f_k(x) \text{ dx} = \int_{\mathbb{R}}\lim_{k\to \infty}f_k(x) \text{ dx} = \int_{\mathbb{R}}f(x) \text{ dx}$.
			\begin{itemize}
				\item Uniform Convergence Theorem: Let $E \subseteq \mathbb{R}$ be L-mesable s.t. $m(E) < \infty$, and $f_n, f : \mathbb{R} \to \mathbb{R}$ L-integrable funcs s.t. $f_n \to f$ unifly on $E$ as $n \to \infty$, Then $f\in \mathcal{L}^1(\mathbb{R})$ and $\lim_{n\to \infty} \int_E f_n(x) \text{ dx} = \int_E \lim_{n\to \infty} f_n (x) \text{ dx}$
			\end{itemize}
			
		\end{mthm}
		
		
		\begin{thm}[The Fundamental Theorem Of Calculus]{thm:ftc}{ }
			Let $a, b\in \mathbb{R}$ with $a < b$ and $f:[a, b] \to \mathbb{R}$ continuous. Then $F(x) := \int_{a}^{x}f(y) \text{ dy}$ is diffable on $[a, b]$ and $F'(x) = f(x)$ for all $x\in [a, b]$
			\begin{itemize}
				\item For $a, b\in \mathbb{R}$, $a < b$, $f: [a, b] \to \mathbb{R}$ cont.sly diffable; $\int_{a}^{b}f'(x) \text{ dx} = f(b) - f(a)$
			\end{itemize}
			
		\end{thm}

		{\tiny\textbf{\underline{Uniform Continuity}}}
		
		
		
		\begin{dfn}[Continuity]{dfn:cont}{ }
		For $I \subseteq \mathbb{R}$ an interval and $f: I \to \mathbb{R}$, $f$ is \textbf{continuous} if for all $x_0\in I$, $\epsilon > 0$, there exists $\delta > 0$ s.t. $\left|f(x) - f(x_0)\right| <\epsilon \forall x$ s.t. $\left|x-x_0\right| < \delta$.
		\begin{itemize}
			\item A cont func. on $[a, b] \subseteq \mathbb{R}$ is bounded and attains a max/min on $[a, b]$.
		\end{itemize}
		
		\end{dfn}
		
		
		\begin{mdfn}[Uniform Continuity]{dfn:unifcont}{ }
		Let $I \subseteq \mathbb{R}$ be an interval and $f: I \to \mathbb{R}$. $f$ is \textbf{uniformly continuous} if $\forall \epsilon > 0$, $\exists \delta > 0$ s.t. $\left|f(x) - f(y)\right| <\epsilon$ $\forall x, y \in I$ where $\left|x-y\right| < \delta$. Note the key difference with regular continuity - here $\delta$ may only depend on $\epsilon$, not $x$.
		
		\end{mdfn}
		
		
		\begin{lma}[Helpful Uniform Continuity Stuff]{lma:unifconthelp}{ }
		\begin{itemize}
			\item If you can bound the deriv of a diffable func, by MVT it is\\
			uniformly continuous (need to prove, for deriv bdd by $M$, use $\delta =\epsilon/M$.).
			\item For all $x\in \mathbb{R}$, $-\frac{\pi}{2} < \arctan(x) < \frac{\pi}{2}$ and $1 \ge \arctan'(x) = \frac{1}{x^2 + 1} > 0$
		\end{itemize}
		
		\end{lma}
		
		
		\begin{thm}[The Heine-Cantor Theorem]{thm:heincan}{ }
		If $a < b$ and $f: [a, b] \to \mathbb{R}$ is continuous, then $f$ is uniformly continuous.
		
		\end{thm}
		
		
		\begin{xmp}[Proving Uniform Continuity]{xmp:proveunifcont}{ }
		Show $f: (0, \infty) \to \mathbb{R}, f(x) := \frac{\sin(x)}{x}$ is uniformly continuous.\\
		$f'(x) = \frac{x\cos(x) - \sin(x)}{x^2} = \frac{x(\cos(x)-1) + (x-\sin(x))}{x^2}$ for $x > 0$.\\
		WLOG assume $x < y$. Apply MVT to $x \mapsto \cos(x)$. For $x > 0 \exists c\in (0, x)$ s.t. $\cos(x) -1 = -\sin(c)x$, so $\left|\cos(x) - 1\right| = \left|-\sin(c)x\right| \le \left|x\right|$.\\
		Apply MVT to $x \mapsto x - \sin(x)$. For $x > 0, \exists d\in (0, x)$ s.t. $x-\sin(x) = (1-\cos(d))x$. As above $\left|1-\cos(d)\right| \le \left|x\right|$, so $\left|x-\sin(x)\right| \le \left|x\right| \left|x\right| = x^2$.\\
		Thus for all $x > 0, \left|f'(x)\right| = \frac{\left|x(\cos(x)-1)\right| + \left|x-\sin(x)\right|}{x^2} \le \frac{x^2 + x^2}{x^2} = 2$
		Then for all $x, y\in (0, \infty) \exists \xi\in (x, y)$ s.t. $\left|f(x) - f(y)\right| = \left|f'(\xi)\right| \left|x-y\right| \le 2\left|x-y\right|$ so let $\delta =\epsilon/2$, then $\left|f(x) - f(y)\right| \le 2\left|x-y\right| < 2 \delta =\epsilon$.
		
		\end{xmp}
		
		
		\begin{xmp}[Disproving Uniform Continuity]{xmp:disproveunifcont}{ }
		Show $f: (0, \infty) \to \mathbb{R}, f(x) = 1/x^2$ is not uniformly continuous.\\
		It suffices to show for all $\delta > 0$ there exist $x, y\in \mathbb{R}$ s.t. $\left|x-y\right| < \delta$ and $\left|f(x) - f(y)\right| \ge 1$. Let $0 < x < y \le 0$. Observe $x + y > x$ and $x^2 y^2 \le x^2$. Then $\frac{1}{x^2} - \frac{1}{y^2} = \frac{y^2 - x^2}{x^2y^2} = \frac{(x-y)(x+y)}{x^2y^2} \ge \frac{y-x}{x} > 0$. Then given $\delta >0$ let $x = \min\{\delta, 1\}/2$ and $y=2x$. \\
		Then $\left|x-y\right| = \min\{\delta, 1\}/2 < \delta$ and $\left|\frac{1}{x^2} - \frac{1}{y^2}\right| \ge \frac{y-x}{x} = \frac{x}{x} = 1$.\\
		Key points - do inequalities in reverse to proving; bound $\left|f(x) - f(y)\right|$ below; $x, y$ may depend on $\delta$; can use any $x, y$ in the domain of $f$.
		
		\end{xmp}
		
		
		\begin{xmp-s}[Diffable On Open Intvl\text{,} Bdd Deriv $\implies$ Unif Cont]{xmp:weirdunifcont}{ }
			Let $I \subseteq \mathbb{R}$ be an open interval and $f: I \to \mathbb{R}$ be differentiable with bounded derivative on $I$. Show $f$ is uniformly continuous on $I$.\\
			$\exists M\in \mathbb{R}$, $\left|f'(x)\right| \le M$ for all $x\in I$. By the MVT, for all $x, y\in \mathbb{R}$, there exists $c$ between $x$ and $y$ s.t. $\left|f(x) - f(y)\right| = \left|f'(c)\right| \left|x-y\right| \le M\left|x-y\right|$. Then given $\epsilon > 0$, let $\delta =\epsilon/M$, and we have unif cont.ty by the definition.
			
		\end{xmp-s}
		
		
		
		{\tiny \textbf{\underline{Uniform Convergence}}}	
		
		
		
		\begin{dfn}[Pointwise Convergence]{dfn:ptwiseconv}{ }
		Let $E \subseteq \mathbb{R}$ be nonempty. A sequence of functions $f_n: E \to \mathbb{R}$ \textbf{converges pointwise on $E$ to $f: E \to \mathbb{R}$} if for all $x\in E$ and $ \epsilon > 0$, there exists $N\in \mathbb{N}$ s.t. $\left|f_n(x) - f(x)\right| <\epsilon$ for all $n \ge N$.\\
		When proving pointwise convergence can just take lim of an expression as $n \to \infty$ to assert pointwise convergence to the lim (if it exists). 
		
		\end{dfn}
		
		
		\begin{thm}[Pointwise Convergence Theorems]{thm:ptwiseconv}{ }
		\begin{itemize}[label=*]
			\item The pointwise limit of bounded functions is not always bounded
			\begin{itemize}
				\item (From exercise; counterxmp $f_n(x) = n/(nx + 1), f(x) = 1/x$ on $(0,1)$)
			\end{itemize}
			\item The ptwise limit of cont funcs is not always cont. $\exists f_n, f: [0,1] \to \mathbb{R}$ s.t.
			\begin{itemize}
				\item $f_n$ is continuous for all $n \in \mathbb{N}$
				\item $f_n \to f$ pointwise on $[0, 1]$ as $n \to \infty$
				\item $f$ is not continuous
				\item Example: $f_n(x) := 0$ if $ 0 \le x \le \frac{1}{2} - \frac{1}{2n}$; $nx - \frac{n-1}{2}$ if $ \frac{1}{2} - \frac{1}{2n} \le x \le \frac{1}{2}+ \frac{1}{2n}$; $1$ if $\frac{1}{2} + \frac{1}{2n} \le x \le 1$
			\end{itemize}
			\item Ptwise lim of diffable funcs not always diffable. $\exists f_n, f: (-1, 1) \to \mathbb{R}$ s.t.:
			\begin{itemize}
				\item $f_n$ is differentiable for all $N \in \mathbb{N}$
				\item $f_n \to f$ pointwise on $(-1, 1)$ as $n \to \infty$
				\item $f$ is not differentiable
				\item Example: Let $f_n(x) := \sqrt{x^2 + \frac{1}{n}}$, limit is not diffable at $0$.
			\end{itemize}
			\item The pointwise limit of derivatives is not always the derivative of the pointwise limit. $\exists f_n, f : (-1, 1) \to \mathbb{R}$ s.t.
			\begin{itemize}
				\item $f_n$ is diffable for all $n \in \mathbb{N}$ and $f$ is diffable
				\item $f_n \to f$ pointwise on $(-1, 1)$
				\item $f'(0) \ne \lim_{n\to \infty}f'_n(0)$, ie $\left(\lim_{n\to \infty f_n}\right)'(0) \ne \lim_{n\to \infty}f_n'(0)$, ie the derivate and ptwise lim operations do not commute.
				\item Example: Let $f_n(x) = \frac{\sin(2 \pi n x)}{2 \pi n}, f'_n = \cos(2\pi n x), f = 0$ on $(-1, 1)$.
			\end{itemize}
		\end{itemize}
		
		\end{thm}
		
		
		\begin{dfn}[Graph and Vertical Neighbourhood]{dfn:graphnvertneigh}{ }
		Let $f: E \to \mathbb{R}$. The \textbf{graph of $f$} is the set $\Gamma(f) : = \left\{(f, f(x) : x\in E)\right\} \subset \mathbb{R}^2$.\\
		For $ \epsilon > 0$, the \textbf{$\epsilon-$vertical neighbourhood of the graph of $f$} is the set \\
		$\mathcal{N}(f,\epsilon) := \left\{(x, y) : x\in E, \left|y-f(x)\right| <\epsilon\right\} \subset \mathbb{R}^2$.
		
		\end{dfn}
		
		
		\begin{mdfn}[Uniform Convergence]{mdfn:unifcon}{ }
		Let $E \subseteq \mathbb{R}$ be nonempty. A sequence of functions $f_n: E \to \mathbb{R}$ \textbf{converges uniformly on $E$ to $f: E \to \mathbb{R}$} if for all $\epsilon > 0$ there exists $N\in \mathbb{N}$ s.t. $\left|f_n(x) - f(x)\right| <\epsilon$ for all $x\in E, n \ge N$.
		\begin{itemize}
			\item Uniform convergence implies pointwise convergence. \\
			For counterexample of converse use $f_n, f: (0, 1]  \to \mathbb{R}$. $f_n(x) := 1 $ if $0 < x < \frac{1}{n}$; $0$ if $\frac{1}{n} \le x \le 1$ and $f(x) :=0$.
		\end{itemize}
		
		\end{mdfn}
		
		
		\begin{mthm}[Uniform Convergence Equivalencies]{thm:unifconv}{ }
		Let $E \subseteq \mathbb{R}$ be nonempty, $f_n : E \to \mathbb{R}$ be a sequence of funcs, and $f: E \to \mathbb{R}$.
		\begin{itemize}
			\item The following are equivalent:
			\begin{enumerate}[label=(\arabic*)]
				\item $f_n \to f$ uniformly on $E$ as $n \to \infty$
				\item $\sup_{x\in E}\left|f_n(x) - f(x)\right| \to 0$ as $n \to \infty$
				\item For all $\epsilon > 0$ there exists $N\in \mathbb{N}$ s.t. $\Gamma(f_n) \subset \mathcal{N}(f,\epsilon)$ for all $n \ge N$.
			\end{enumerate}
			Let $f_n \to f$ uniformly on $E$ as $n \to \infty$. If $(x_n)_{n\in \mathbb{N}}$ is a real sequence with $x_n\in E$ for all $n\in \mathbb{N}$, then $\left|f_n(x_n) - f(x_n)\right| \to 0$ as $n \to \infty$.
		\end{itemize}
		
		\end{mthm}

		\vspace{1pt}
		\hrule	
		\vspace{-6pt}
		\begin{proof}
			Let $\epsilon > 0$. Since $f_n \to f$ unifly on $I$, $\exists N\in \mathbb{N}$ s.t. $\left|f_n(x) - f(x)\right| <\epsilon/3$ for all $x\in I, n \ge N$. Then by the TI, $\left|f(x) - f(x_0)\right| \le \left|f(x) - f_N(x)\right| + \left|f_N(x) - f_N(x_0)\right| + \left|f_N(x_0) -f(x_0)\right| <\epsilon$. Thus $f$ is cont. at $x_0\in I$.
		\end{proof}
		
		\end{thm}

		\end{enumerate}
		Then there exists diffable $f : I \to \mathbb{R}$ such that $\sum_{n=1}^{\infty}f_n$ converges uniformly to $f$ on $I$, and $f'(x) = \left(\sum_{n=1}^{\infty}f_n\right)'(x) = \sum_{n=1}^{\infty}f_n'(x)$ for all $x\in I$.

		\begin{thm}[Weierstrass M-Test]{thm:weierm}{ }
		Let $E \subseteq \mathbb{R}$ be $\ne \emptyset$, $f_n: E \to \mathbb{R}$, and $(M_n)_{n\in \mathbb{N}}$ a non-negative seq satisfying:
		\begin{enumerate}[label=(\arabic*)]
			\item $\left|f_n(x)\right| \le M_n$ for all $x\in E, n\in \mathbb{N}$
			\vspace{1pt}
			\item $\sum_{n=1}^{\infty}M_n$ converges
		\end{enumerate}
		Then $\sum_{n=1}^{\infty}f_n$ converges absolutely and uniformly on $E$.
		\vspace{1pt}
		\hrule	
		\vspace{-6pt}
		\begin{proof}
			Let $\epsilon > 0$. By the CC for series, $\exists N\in \mathbb{N}$ s.t. $\sum_{k=m+1}^{n} M_k <\epsilon$ for $n \ge m \ge N$. Thus by hypothesis, $\left|\sum_{k=m+1}^{n} f_k(x)\right| \le \sum_{k=m+1}^{n}\left|f_k(x)\right| \le \sum_{k=m+1}^{n}M_k <\epsilon$ for all $x\in E, n > m \ge N$. Thus the partial sums of $\sum_{n=1}^{\infty}f_n$ are unifly Cauchy, so $\sum_{n=1}^{\infty}f_n$ converges unifly. Moreover, by the above, the partial sums of $\sum_{n=1}^{\infty}\left|f_n\right|$ are unifly Cauchy, so the series converges absolutely.
		\end{proof}
		
		\end{thm}

		\begin{dfn}[Power Series]{dfn:powseries}{ }
			Let $(a_n)_{n\in \mathbb{N}}$ be a real sequence and $c\in \mathbb{R}$. A \textbf{power series centred at $c$} is a series of the form $\sum_{n=0}^{\infty}a_n(x-c)^n$.
		
		\end{dfn}
		
		
		\begin{dfn}[Radius Of Convergence $R$]{dfn:radofconv}{ }
			Let $(a_n)_{n\in \mathbb{N}}$ be a real seq and $c\in \mathbb{R}$. Then one of the following holds:
			\begin{enumerate}[label=(\arabic*)]
				\item $\exists R > 0, R\in \mathbb{R}$ s.t. $\sum_{n=0}^{\infty}a_n(x-c)^n$ cnvges absolutely if $\left|x - c\right| < R$ and diverges if $\left|x-c\right|> R$. Here $R = \sup\{r > 0 : (a_nr^n)_{n\in \mathbb{N}_0} \text{is bdd}\}$
				\item $\sum_{n=0}^{\infty}a_n(x-c)^n$ converges only when $x = c$. Here $R = 0$
				\item $\sum_{n=0}^{\infty}a_n(x-c)^n$ converges absolutely for all $x\in \mathbb{R}$. Here $R = \infty$
			\end{enumerate}.
		
		\end{dfn}
		
		
		\begin{thm}[On Termwise Differentiation Of Power Series]{thm:contndiffpwrseries}{ }
		Let $\sum_{n=0}^{\infty}a_n(x-c)^n$ have radius of convergence $0 < R \le \infty$. Then $f(x) = \sum_{n=0}^{\infty}a_n(x-c)^n$ for $x\in (c-R, c+R)$ is differentiable on $(c-R, c+R)$; and $f'(x) = \sum_{n=1}^{\infty}na_n(x-c)^{n-1}$ for all $x\in (c-R, c+R)$.
		\begin{itemize}
			\item Given the above, $f(x) := \sum_{n=0}^{\infty}a_n(x-c)^n$  is \textbf{infinitely differentiable} on $(c-R, c+R)$, and $a_k = \frac{f^{(k)}(c)}{k!}$ for all $k\in \mathbb{N}_0$.
			\item For $0 < r < R$, both series converge absolutely and unifly on $[c-r, c+r]$
			\item For any real sequence $(a_n)_{n\in \mathbb{N}_0}$ and $c\in \mathbb{R}$, $\sum_{n=0}^{\infty}a_n(x-c)^n$ and $\sum_{n=1}^{\infty}na_n(x-c)^{n-1}$ have the same radius of convergence
		\end{itemize}
		
		\end{thm}
		
		
		\begin{mthm}[Uniqueness Of Power Series]{thm:uniqpwrsers}{ }
		Power series are unique - For $\sum_{n=0}^{\infty}a_n(x-c)^n$ and $\sum_{n=0}^{\infty}b_n(x-c)^n$ with RoCs $0 < R, S, \le \infty$, if $\exists 0 < r < \min{R, S}$ s.t. $\sum_{n=0}^{\infty} a_n(x-c)^n = \sum_{n=0}^{\infty}b_n (x-c)^n$ for all $x\in \mathbb{R}$ s.t. $\left|x-c\right| < r$, then $a_n = b_n$ for all $n\in \mathbb{N}$.
		\vspace{1pt}
		\hrule	
		\vspace{-6pt}
		\begin{proof}
			Let $I = (c-r, c+r)$, and $f: I \to \mathbb{R}, f(x) := \sum_{n=0}^{\infty}a_n(x-c)^n$ for $x\in I$. We know $f$ is infinitely diffable on $I$, and $a_n = \frac{f^{(n)}(c)}{n!}$ for $n\in \mathbb{N}_0$. However, $f(x) = \sum_{n=0}^{\infty}b_n(x-c)^n$ for $x\in I$, and $b_n = \frac{f^{(n)}(c)}{n!}$ for all $n\in \mathbb{N}_0$. Thus $a_n = b_n$ for all $n\in \mathbb{N}_0$.
		\end{proof}
		
		\end{mthm}

Taking the infimum we see $m(E + x) \le m(E) +\epsilon$, since $\epsilon > 0$ arbitrary $m(E+x) \le m(E)$.\\
			We can apply the same inequality in reverse (since $E = (E+x) -x$ is a translate of $E+x$. Thus $m(E) \le m(E+x) \le m(E)$, so $m(E) = m(E+x)$.
		
		\end{xmp}

			\begin{itemize}
				\item \textit{Sequences MCT}: If a real sequence $(a_n)_{n \in \mathbb{N}}$ is monotone non-decreasing and bdd above; or monotone non-increasing and bdd below; it converges.
				\item If a seq converges to $L$, all its subseqs also converge to $L$ (so to disprove convergence find two subseqs converging to different limits).

		Use the DCT to evaluate $\lim_{n\to \infty} \int_{0}^{1}\frac{1}{(1+x/n)^n x^{1/n}} \text{ dx}$.\\
		Let $f_n: \mathbb{R} \to \mathbb{R}$ be $f_n(x) := (1+x/n)^{-n} x^{-1/n} \chi_{(0, \infty)}(x)$ for all $x\in \mathbb{R}$. We claim $f_n$ is L-mesable. Since is is not cont. on $\mathbb{R}$, given $\lambda\in \mathbb{R}$ let $E_\lambda := \{x\in \mathbb{R} : f_n(x) > \lambda\}$. Then if $\lambda < 0$, $E_\lambda = \mathbb{R}$ which is L-mesable. If $\lambda \ge 0$, $E_\lambda = \{x\in (0, \infty) : (1+ x/n)^{-n}x^{-1/n} > \lambda\}$. $g_n : (0, \infty) \to \mathbb{R}$, $g_n(x) := (1+ x/n)^{-n}x^{-1/n}$ for $x > 0$ is a cont. func over an open intvl, so $E_\lambda$ is an open set and thus L-mesable. Thus $f_n(x)$ is L-mesable.\\
		We know $(1 + 1/x)^{-n} \to e^{-x}$ ptwise on $[0, \infty)$, and $x^{-1/n} \to 1$ ptwise on $(0, \infty)$, as $n \to \infty$. Thus $f(x) := e^{-x} \chi_{(0, \infty)}(x)$ for $x\in \mathbb{R}$ is the ptwise lim of $f_n$ on $\mathbb{R}$.\\
		By the binomial theorem, for $x > 0$, $(1 + \frac{x}{n})^n \ge 1 + n \cdot x/n + \binom{n}{2} \left(\frac{x}{n}\right)^2 = 1 + x + \frac{n(n-1)}{2n^2} \cdot x^2$. If $n \ge 2$, $ n-1 \ge n/2$, so $(1 + x/n)^n \ge 1 + x^2/4$. It follows from these that $(1 + x/n)^{-n} \le \frac{4}{4+x^2}$. Moreover, $x^{1/n} \ge 1$ for $x \ge 1$ and $x^{1/n} \ge x^{1/2}$ for $0 < x \le 1$ and $n \ge 2$. Combining all these observations:\\
		$\left|f_n(x)\right| \le \ell(x) := \frac{4}{4 + x^2}\chi_{[1, \infty)}(x) + \frac{1}{x^{1/2}}\chi_{(0,1)}(x)$ for all $x\in \mathbb{R}, n \ge 2$, and $\ell(x)$ is non-neg and L-mesable.\\
		We claim $\ell$ is L-intable. Indeed by monotonicity of the integral we have $\int_{1}^{\infty} \frac{4}{4+x^2} \chi_{[1, \infty)}(x) \text{ dx} \le \int_{1}^{\infty} \frac{4}{x^2} \text{ dx}$, where the RHS is finite (proof left as an exercise). Moreover define $h: (0, \infty) \to \mathbb{R}$ by $h(x) := x^{-1/2}$ for $x > 0$. Then $H: (0, \infty) \to \mathbb{R}, H(x) := 2x^{1/2}$ for $x> 0$ is a primitive for $h$. By the FTC, for $\epsilon > 0$, $\int_{eps}^{1} x^{-1/2} \text{ dx} = \left[2x^{1/2}\right]_\epsilon^1 = 2 - 2e^{1/2}$.\\
		By the MCT, $\int_{\mathbb{R}} \frac{1}{x^{1/2}}\chi_{(0,1)}(x) \text{ dx} = \int_{0}^{1} \frac{1}{x^{1/2}} \text{ dx} = \lim_{\epsilon\to 0^+} \int_{\epsilon}^{1}\frac{1}{x^{1/2}}dx = \lim_{\epsilon\to 0^+} 2 - 2\epsilon^{1/2} = 2$.\\
		Combining these observations with the linearity of the integral, $\int_{\mathbb{R}} \ell(x) \text{ dx} < \infty$, so $\ell$ is L-intable. Thus we may apply the DCT and conclude:\\
		$\lim_{n\to \infty} \int_{0}^{\infty} \frac{1}{(1+x/n)^n x^{1/n}} \text{ dx} = \lim_{n\to \infty} \int_{\mathbb{R}}f_n(x) \text{ dx} = \int_{\mathbb{R}} \lim_{n\to \infty} f_n(x) \text{ dx} = \int_{0}^{\infty} e^{-x} \text{ dx} = 1$.
		
		\end{xmp}
